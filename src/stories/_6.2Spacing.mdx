export const spacing = {
  values: [  
    { key: '00', name: 'spacing-00', value: '0px' },
    { key: '01', name: 'spacing-01', value: '2px' },
    { key: '02', name: 'spacing-02', value: '4px' },
    { key: '03', name: 'spacing-03', value: '8px' },
    { key: '04', name: 'spacing-04', value: '12px' },
    { key: '05', name: 'spacing-05', value: '16px' },
    { key: '06', name: 'spacing-06', value: '20px' },
    { key: '07', name: 'spacing-07', value: '24px' },
    { key: '08', name: 'spacing-08', value: '32px' },
    { key: '09', name: 'spacing-09', value: '40px' },
    { key: '10', name: 'spacing-10', value: '48px' },
    { key: '11', name: 'spacing-11', value: '56px' }
  ]
};

export const TokenSearchBar = ({ value, onChange, placeholder, matches, total }) => (
  <div className="token-search-wrap">
    <div className="token-search">
      <span className="token-search-icon" aria-hidden="true">{'\u{1F50D}'}</span>
      <input
        type="text"
        className="token-search-input"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        placeholder={placeholder || 'Search tokens...'}
      />
      {value && (
        <button
          type="button"
          className="token-search-clear"
          onClick={() => onChange('')}
          aria-label="Clear search"
        >
          {'✕'}
        </button>
      )}
    </div>
    {value && (
      <div className="token-search-meta">
        {matches} of {total} tokens match "<b>{value}</b>"
      </div>
    )}
  </div>
);

export const PaddingDemo = () => {
  const [query, setQuery] = React.useState('');
  const q = query.trim().toLowerCase();

  const matches = (...fields) =>
    !q || fields.some((f) => f != null && String(f).toLowerCase().includes(q));

  const filterByPrefix = (prefix) =>
    spacing.values.filter((item) =>
      matches(`${prefix}-${item.key}`, item.key, item.name, item.value)
    );

  const all = filterByPrefix('global-p');
  const px = filterByPrefix('global-px');
  const py = filterByPrefix('global-py');

  const sideGroups = spacing.values
    .map((item) => {
      const sides = [
        { prefix: 'global-pt', style: { paddingTop: item.value }, label: 'Top Padding' },
        { prefix: 'global-pr', style: { paddingRight: item.value }, label: 'Right Padding' },
        { prefix: 'global-pb', style: { paddingBottom: item.value }, label: 'Bottom Padding' },
        { prefix: 'global-pl', style: { paddingLeft: item.value }, label: 'Left Padding' }
      ].filter((s) =>
        matches(`${s.prefix}-${item.key}`, item.key, item.name, item.value)
      );
      return { item, sides };
    })
    .filter((g) => g.sides.length > 0);

  const sideMatchCount = sideGroups.reduce((n, g) => n + g.sides.length, 0);

  const total = spacing.values.length * 7;
  const matchCount = all.length + px.length + py.length + sideMatchCount;

  const hasAny = matchCount > 0;

  return (
    <>
      <TokenSearchBar
        value={query}
        onChange={setQuery}
        placeholder="Search padding tokens (e.g. global-px-05, 16px, spacing-05)..."
        matches={matchCount}
        total={total}
      />

      {!hasAny && (
        <div className="token-search-empty">
          No padding tokens match "<b>{query}</b>".
        </div>
      )}

      <h2>Padding Utilities</h2>
      <div className="sb-demo-section">
        {all.length > 0 && (
          <div className="sb-utility-group">
            <h3>All Sides Padding</h3>
            <div className="sb-demo-box">
              {all.map((item) => (
                <div
                  key={item.key}
                  className="sb-demo-content"
                  style={{
                    padding: item.value,
                    background: 'rgb(162 206 130)',
                    marginBottom: '10px'
                  }}
                >
                  <div className="padding-demo">
                    <code>.global-p-{item.key}</code>
                    <p>Padding: {item.name} ({item.value})</p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {px.length > 0 && (
          <div className="sb-utility-group">
            <h3>Horizontal Padding (px)</h3>
            <div className="sb-demo-box">
              {px.map((item) => (
                <div
                  key={item.key}
                  className="sb-demo-content"
                  style={{
                    paddingLeft: item.value,
                    paddingRight: item.value,
                    background: 'rgb(162 206 130)',
                    marginBottom: '10px'
                  }}
                >
                  <div className="padding-demo">
                    <code>.global-px-{item.key}</code>
                    <p>Horizontal Padding: {item.name} ({item.value})</p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {py.length > 0 && (
          <div className="sb-utility-group">
            <h3>Vertical Padding (py)</h3>
            <div className="sb-demo-box">
              {py.map((item) => (
                <div
                  key={item.key}
                  className="sb-demo-content"
                  style={{
                    paddingTop: item.value,
                    paddingBottom: item.value,
                    background: 'rgb(162 206 130)',
                    marginBottom: '10px'
                  }}
                >
                  <div className="padding-demo">
                    <code>.global-py-{item.key}</code>
                    <p>Vertical Padding: {item.name} ({item.value})</p>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {sideGroups.length > 0 && (
          <div className="sb-utility-group">
            <h3>Individual Padding Sides</h3>
            <div className="sb-demo-box">
              {sideGroups.map(({ item, sides }) => (
                <div key={item.key} className="sb-demo-content">
                  {sides.map((s) => (
                    <div
                      key={s.prefix}
                      style={{
                        ...s.style,
                        background: 'rgb(162 206 130)',
                        marginBottom: '10px'
                      }}
                    >
                      <div className="padding-demo">
                        <code>.{s.prefix}-{item.key}</code>
                        <p>{s.label}: {item.name} ({item.value})</p>
                      </div>
                    </div>
                  ))}
                </div>
              ))}
            </div>
          </div>
        )}
      </div>
    </>
  );
};

  <div className="sb-section">

    <h2>Usage Examples</h2>
    <div className="sb-code-block">

      <div class="global-py-05">Vertical padding 16px</div>

  </div>
</div>
