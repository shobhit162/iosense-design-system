export const TokenSearchBar = ({ value, onChange, placeholder, matches, total }) => (
  <div className="token-search-wrap">
    <div className="token-search">
      <span className="token-search-icon" aria-hidden="true">{'\u{1F50D}'}</span>
      <input
        type="text"
        className="token-search-input"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        placeholder={placeholder || 'Search tokens...'}
      />
      {value && (
        <button
          type="button"
          className="token-search-clear"
          onClick={() => onChange('')}
          aria-label="Clear search"
        >
          {'✕'}
        </button>
      )}
    </div>
    {value && (
      <div className="token-search-meta">
        {matches} of {total} tokens match "<b>{value}</b>"
      </div>
    )}
  </div>
);

export const SpacingDemo = () => {
  const [query, setQuery] = React.useState('');
  const q = query.trim().toLowerCase();

  const matches = (...fields) =>
    !q || fields.some((f) => f != null && String(f).toLowerCase().includes(q));

  const filtered = spacing.values.filter((item) =>
    matches(item.key, item.name, item.value)
  );

  return (
    <>
      <TokenSearchBar
        value={query}
        onChange={setQuery}
        placeholder="Search spacing tokens (e.g. 16px, spacing-05, 05)..."
        matches={filtered.length}
        total={spacing.values.length}
      />

      {filtered.length === 0 ? (
        <div className="token-search-empty">
          No spacing tokens match "<b>{query}</b>".
        </div>
      ) : (
        <div className="sb-spacing-scale">
          {filtered.map((item) => (
            <div className="sb-scale-item" key={item.key}>
              <div className="sb-scale-visual">
                <div
                  className="sb-scale-box"
                  style={{
                    width: item.value,
                    height: item.value,
                    backgroundColor: '#E5E7EB'
                  }}
                />
              </div>
              <div className="sb-scale-info">
                <code>{item.value}</code>
                <span>{item.name}</span>
              </div>
            </div>
          ))}
        </div>
      )}
    </>
  );
};
