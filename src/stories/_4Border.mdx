export const borders = {
  borderWidth: {
    'none': { name: 'global-bw-none', value: '0px' },
    'thinner': { name: 'global-bw-thinner', value: '0.5px' },
    'thin': { name: 'global-bw-thin', value: '1px' },
    'thick': { name: 'global-bw-thick', value: '1.5px' },
    'thicker': { name: 'global-bw-thicker', value: '2px' }
  },
  borderStyles: {
    'solid': { name: 'global-bs-solid', value: 'solid' },
    'dashed': { name: 'global-bs-dashed', value: 'dashed' },
    'dotted': { name: 'global-bs-dotted', value: 'dotted' },
    'double': { name: 'global-bs-double', value: 'double' },
    'none': { name: 'global-bs-none', value: 'none' }
  },
  borderRadius: {
    'none': { name: 'global-br-none', value: '0px' },
    'xsmall': { name: 'global-br-xsmall', value: '1px' },
    'small': { name: 'global-br-small', value: '2px' },
    'medium': { name: 'global-br-medium', value: '4px' },
    'large': { name: 'global-br-large', value: '8px' },
    'xlarge': { name: 'global-br-xlarge', value: '12px' },
    '2xlarge': { name: 'global-br-2xlarge', value: '16px' },
    'max': { name: 'global-br-max', value: '1000px' }
  }
};

export const TokenSearchBar = ({ value, onChange, placeholder, matches, total }) => (
  <div className="token-search-wrap">
    <div className="token-search">
      <span className="token-search-icon" aria-hidden="true">{'\u{1F50D}'}</span>
      <input
        type="text"
        className="token-search-input"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        placeholder={placeholder || 'Search tokens...'}
      />
      {value && (
        <button
          type="button"
          className="token-search-clear"
          onClick={() => onChange('')}
          aria-label="Clear search"
        >
          {'✕'}
        </button>
      )}
    </div>
    {value && (
      <div className="token-search-meta">
        {matches} of {total} tokens match "<b>{value}</b>"
      </div>
    )}
  </div>
);

export const BorderDemo = () => {
  const [query, setQuery] = React.useState('');
  const q = query.trim().toLowerCase();

  const matches = (...fields) =>
    !q || fields.some((f) => f != null && String(f).toLowerCase().includes(q));

  const filterBase = (group) =>
    Object.entries(group).filter(([key, { name, value }]) =>
      matches(key, name, value)
    );

  const widthEntries = filterBase(borders.borderWidth);
  const styleEntries = filterBase(borders.borderStyles);
  const radiusEntries = filterBase(borders.borderRadius);

  const sidePrefixes = ['global-bw-t', 'global-bw-r', 'global-bw-b', 'global-bw-l'];
  const cornerPrefixes = ['global-br-tl', 'global-br-tr', 'global-br-bl', 'global-br-br'];
  const combinedPrefixes = ['global-br-t', 'global-br-r', 'global-br-b', 'global-br-l'];

  const buildRows = (group, prefixes) =>
    Object.entries(group)
      .map(([key, entry]) => {
        const cells = prefixes
          .map((prefix) => ({
            prefix,
            key,
            value: entry.value,
            className: `${prefix}-${key}`
          }))
          .filter((c) => matches(c.className, c.key, c.value));
        return { key, entry, cells };
      })
      .filter((row) => row.cells.length > 0);

  const sideRows = buildRows(borders.borderWidth, sidePrefixes);
  const cornerRows = buildRows(borders.borderRadius, cornerPrefixes);
  const combinedRows = buildRows(borders.borderRadius, combinedPrefixes);

  const sideMatchCount = sideRows.reduce((n, r) => n + r.cells.length, 0);
  const cornerMatchCount = cornerRows.reduce((n, r) => n + r.cells.length, 0);
  const combinedMatchCount = combinedRows.reduce((n, r) => n + r.cells.length, 0);

  const totalTokens =
    Object.keys(borders.borderWidth).length +
    Object.keys(borders.borderStyles).length +
    Object.keys(borders.borderRadius).length +
    Object.keys(borders.borderWidth).length * sidePrefixes.length +
    Object.keys(borders.borderRadius).length * cornerPrefixes.length +
    Object.keys(borders.borderRadius).length * combinedPrefixes.length;

  const matchCount =
    widthEntries.length +
    styleEntries.length +
    radiusEntries.length +
    sideMatchCount +
    cornerMatchCount +
    combinedMatchCount;

  const cornerStyleFor = (prefix, value) => {
    switch (prefix) {
      case 'global-br-tl':
        return { borderTopLeftRadius: value };
      case 'global-br-tr':
        return { borderTopRightRadius: value };
      case 'global-br-bl':
        return { borderBottomLeftRadius: value };
      case 'global-br-br':
        return { borderBottomRightRadius: value };
      default:
        return {};
    }
  };

  const combinedStyleFor = (prefix, value) => {
    switch (prefix) {
      case 'global-br-t':
        return { borderTopLeftRadius: value, borderTopRightRadius: value };
      case 'global-br-r':
        return { borderTopRightRadius: value, borderBottomRightRadius: value };
      case 'global-br-b':
        return { borderBottomLeftRadius: value, borderBottomRightRadius: value };
      case 'global-br-l':
        return { borderTopLeftRadius: value, borderBottomLeftRadius: value };
      default:
        return {};
    }
  };

  const sideStyleFor = (prefix, value) => {
    switch (prefix) {
      case 'global-bw-t':
        return { borderTop: `${value} solid #666` };
      case 'global-bw-r':
        return { borderRight: `${value} solid #666` };
      case 'global-bw-b':
        return { borderBottom: `${value} solid #666` };
      case 'global-bw-l':
        return { borderLeft: `${value} solid #666` };
      default:
        return {};
    }
  };

  return (
    <div className="sb-border-demo">
      <TokenSearchBar
        value={query}
        onChange={setQuery}
        placeholder="Search border tokens (e.g. thin, 2px, global-br-large, global-bw-r-none)..."
        matches={matchCount}
        total={totalTokens}
      />

      {matchCount === 0 && (
        <div className="token-search-empty">
          No border tokens match "<b>{query}</b>".
        </div>
      )}

      {widthEntries.length > 0 && (
        <>
          <h2>Border Width</h2>
          <div className="sb-demo-section">
            {widthEntries.map(([key, { name, value }]) => (
              <div className="sb-demo-item" key={key}>
                <div className="sb-demo-info">
                  <code>{name}</code>
                  <span>{value}</span>
                </div>
                <div
                  className="sb-demo-box"
                  style={{ border: `${value} solid #666` }}
                />
              </div>
            ))}
          </div>
        </>
      )}

      {sideRows.length > 0 && (
        <>
          <h3>Individual Border Sides</h3>
          <div className="sb-demo-section">
            {sideRows.map(({ key, cells }) => (
              <div key={key} className="sb-demo-row">
                {cells.map((c) => (
                  <div key={c.prefix} className="sb-demo-info">
                    <code>{c.className}</code>
                    <div
                      className="sb-demo-box"
                      style={sideStyleFor(c.prefix, c.value)}
                    />
                  </div>
                ))}
              </div>
            ))}
          </div>
        </>
      )}

      {styleEntries.length > 0 && (
        <>
          <h2>Border Styles</h2>
          <div className="sb-demo-section">
            {styleEntries.map(([key, { name, value }]) => (
              <div className="sb-demo-item" key={key}>
                <div className="sb-demo-info">
                  <code>{name}</code>
                  <span>{value}</span>
                </div>
                <div
                  className="sb-demo-box"
                  style={{ border: `2px ${value} #666` }}
                />
              </div>
            ))}
          </div>
        </>
      )}

      {radiusEntries.length > 0 && (
        <>
          <h2>Border Radius</h2>
          <div className="sb-demo-section">
            {radiusEntries.map(([key, { name, value }]) => (
              <div className="sb-demo-item" key={key}>
                <div className="sb-demo-info">
                  <code>{name}</code>
                  <span>{value}</span>
                </div>
                <div
                  className="sb-demo-box"
                  style={{ border: '2px solid #666', borderRadius: value }}
                />
              </div>
            ))}
          </div>
        </>
      )}

      {cornerRows.length > 0 && (
        <>
          <h3>Individual Corner Radius</h3>
          <div className="sb-demo-section">
            {cornerRows.map(({ key, cells }) => (
              <div key={key} className="sb-demo-row">
                {cells.map((c) => (
                  <div key={c.prefix} className="sb-demo-info">
                    <code>{c.className}</code>
                    <div
                      className="sb-demo-box"
                      style={{
                        border: '2px solid #666',
                        ...cornerStyleFor(c.prefix, c.value)
                      }}
                    />
                  </div>
                ))}
              </div>
            ))}
          </div>
        </>
      )}

      {combinedRows.length > 0 && (
        <>
          <h3>Combined Corner Radius</h3>
          <div className="sb-demo-section">
            {combinedRows.map(({ key, cells }) => (
              <div key={key} className="sb-demo-row">
                {cells.map((c) => (
                  <div key={c.prefix} className="sb-demo-info">
                    <code>{c.className}</code>
                    <div
                      className="sb-demo-box"
                      style={{
                        border: '2px solid #666',
                        ...combinedStyleFor(c.prefix, c.value)
                      }}
                    />
                  </div>
                ))}
              </div>
            ))}
          </div>
        </>
      )}
    </div>
  );
};

<div className="sb-container">
  <div className="sb-section-title">
    # Border
    These tokens should be used for consistent border styling across applications.
  </div>

  <div className="sb-section">
    <BorderDemo />

    <h2>Usage Examples</h2>
    <div className="sb-code-block">
      ```html
      <!-- Basic border -->
      <div class="global-bw-thin global-bs-solid">
        1px solid border
      </div>

      <!-- Rounded corners -->
      <div class="global-bw-thin global-bs-solid global-br-medium">
        1px solid border with 4px border radius
      </div>

      <!-- Individual sides -->
      <div class="global-bw-b-thick">
        1.5px bottom border only
      </div>

      <!-- Individual corners -->
      <div class="global-br-tl-large global-br-br-large">
        8px radius on top-left and bottom-right corners
      </div>
      ```
    </div>
  </div>
</div>
