{/* Helper function to convert hex to rgb */}
export const hexToRgb = (hex) => {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result ? {
    r: parseInt(result[1], 16),
    g: parseInt(result[2], 16),
    b: parseInt(result[3], 16)
  } : null;
};

{/* Helper function to convert hex-based rgba to valid CSS rgba */}
export const convertHexRgba = (value) => {
  if (value.includes('rgba(#')) {
    const matches = value.match(/rgba\(#([A-Fa-f0-9]{6}),\s*([\d.]+)\)/);
    if (matches) {
      const hex = matches[1];
      const alpha = matches[2];
      const rgb = hexToRgb(`#${hex}`);
      return rgb ? `rgba(${rgb.r}, ${rgb.g}, ${rgb.b}, ${alpha})` : value;
    }
  }
  return value;
};

export const TokenSearchBar = ({ value, onChange, placeholder, matches, total }) => (
  <div className="token-search-wrap">
    <div className="token-search">
      <span className="token-search-icon" aria-hidden="true">{'\u{1F50D}'}</span>
      <input
        type="text"
        className="token-search-input"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        placeholder={placeholder || 'Search tokens...'}
      />
      {value && (
        <button
          type="button"
          className="token-search-clear"
          onClick={() => onChange('')}
          aria-label="Clear search"
        >
          {'✕'}
        </button>
      )}
    </div>
    {value && (
      <div className="token-search-meta">
        {matches} of {total} tokens match "<b>{value}</b>"
      </div>
    )}
  </div>
);

export const BaseNewColorsDemo = () => {
  const [query, setQuery] = React.useState('');
  const q = query.trim().toLowerCase();

  const matches = (...fields) =>
    !q || fields.some((f) => f != null && String(f).toLowerCase().includes(q));

  const filterShades = (shades) =>
    Object.entries(shades).filter(([shade, s]) => matches(shade, s.name, s.value));

  const filterGroups = (groups) =>
    Object.entries(groups)
      .map(([name, shades]) => [name, filterShades(shades)])
      .filter(([, entries]) => entries.length > 0);

  const chromaticGroups = filterGroups(colors.chromatic);
  const neutralGroups = filterGroups(colors.neutral);

  const countAll = (groups) =>
    Object.values(groups).reduce((n, shades) => n + Object.keys(shades).length, 0);
  const countMatches = (groupEntries) =>
    groupEntries.reduce((n, [, entries]) => n + entries.length, 0);

  const total = countAll(colors.chromatic) + countAll(colors.neutral);
  const matchCount = countMatches(chromaticGroups) + countMatches(neutralGroups);

  return (
    <>
      <TokenSearchBar
        value={query}
        onChange={setQuery}
        placeholder="Search color tokens (e.g. --global-chromatic-brand-500, #F04438)..."
        matches={matchCount}
        total={total}
      />

      {matchCount === 0 && (
        <div className="token-search-empty">
          No color tokens match "<b>{query}</b>".
        </div>
      )}

      {(chromaticGroups.length > 0 || neutralGroups.length > 0) && (
        <div className="sb-section">
          <div className="sb-colors-demo">
            <h2>Base Classes (Newly added as on Dec 2025)</h2>

            {chromaticGroups.length > 0 && (
              <>
                <h3>Chromatic Colors</h3>
                {chromaticGroups.map(([colorName, entries]) => (
                  <div key={colorName} className="color-section">
                    <h4>{colorName.charAt(0).toUpperCase() + colorName.slice(1)}</h4>
                    <div className="color-grid">
                      {entries.map(([shade, { name, value }]) => (
                        <div key={shade} className="color-item">
                          <div
                            className="color-preview"
                            style={{ backgroundColor: convertHexRgba(value) }}
                          ></div>
                          <div className="color-info">
                            <div className="color-variables">
                              <h4>Variable</h4>
                              <code>{name}</code>
                            </div>
                            <span>{value}</span>
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                ))}
              </>
            )}

            {neutralGroups.length > 0 && (
              <>
                <h3>Neutral Colors</h3>
                {neutralGroups.map(([colorName, entries]) => (
                  <div key={colorName} className="color-section">
                    <h4>{colorName.charAt(0).toUpperCase() + colorName.slice(1)}</h4>
                    <div className="color-grid">
                      {entries.map(([shade, { name, value }]) => (
                        <div key={shade} className="color-item">
                          <div
                            className="color-preview"
                            style={{ backgroundColor: convertHexRgba(value) }}
                          ></div>
                          <div className="color-info">
                            <div className="color-variables">
                              <h4>Variable</h4>
                              <code>{name}</code>
                            </div>
                            <span>{value}</span>
                          </div>
                        </div>
                      ))}
                    </div>
                  </div>
                ))}
              </>
            )}
          </div>
        </div>
      )}
    </>
  );
};
