export const CopyButton = ({ text }) => {
  const [buttonText, setButtonText] = React.useState('Copy');
  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(text);
      setButtonText('Copied!');
      setTimeout(() => {
        setButtonText('Copy');
      }, 1000);
    } catch (err) {
      console.error('Failed to copy:', err);
      setButtonText('Failed');
      setTimeout(() => {
        setButtonText('Copy');
      }, 1000);
    }
  };

  return (
    <button 
      onClick={handleCopy}
      style={{
        cursor: 'pointer',
        position: 'absolute',
        right: '8px',
        top: '50%',
        transform: 'translateY(-50%)',
        padding: '4px 8px',
        fontSize: '12px',
        color: buttonText === 'Copied!' ? 'rgb(200 16 16)' : buttonText === 'Failed' ? '#f00' : '#666',
        transition: 'color 0.2s ease',
      }}
    >
      {buttonText}
    </button>
  );
};

export const TokenSearchBar = ({ value, onChange, placeholder, matches, total }) => (
  <div className="token-search-wrap">
    <div className="token-search">
      <span className="token-search-icon" aria-hidden="true">{'\u{1F50D}'}</span>
      <input
        type="text"
        className="token-search-input"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        placeholder={placeholder || 'Search tokens...'}
      />
      {value && (
        <button
          type="button"
          className="token-search-clear"
          onClick={() => onChange('')}
          aria-label="Clear search"
        >
          {'✕'}
        </button>
      )}
    </div>
    {value && (
      <div className="token-search-meta">
        {matches} of {total} tokens match "<b>{value}</b>"
      </div>
    )}
  </div>
);

export const TypeScaleDemo = () => {
  const [query, setQuery] = React.useState('');
  const q = query.trim().toLowerCase();

  const matches = (...fields) =>
    !q || fields.some((f) => f != null && String(f).toLowerCase().includes(q));

  const filterGroup = (group) =>
    Object.entries(group).filter(([className, styles]) =>
      matches(
        className,
        styles.fontSize.class,
        styles.lineHeight.class,
        styles.letterSpacing.class,
        styles.fontWeight.class,
        styles.paragraphSpacing.class
      )
    );

  const display = filterGroup(typography.typeScale.display);
  const heading = filterGroup(typography.typeScale.heading);
  const body = filterGroup(typography.typeScale.body);

  const total =
    Object.keys(typography.typeScale.display).length +
    Object.keys(typography.typeScale.heading).length +
    Object.keys(typography.typeScale.body).length;

  const matchCount = display.length + heading.length + body.length;

  const renderGroup = (title, entries, example) =>
    entries.length > 0 && (
      <>
        <h3>{title}</h3>
        <div className="sb-type-scale-section">
          {entries.map(([className, styles]) => (
            <div className="sb-type-scale-item" key={className}>
              <div className="sb-type-scale-info">
                <div className="code-wrapper">
                  <code>{className}</code>
                  <CopyButton text={className} />
                </div>
                <small>
                  Font Size Class: {styles.fontSize.class}
                  <br />
                  Line Height Class: {styles.lineHeight.class}
                  <br />
                  Letter Spacing: {styles.letterSpacing.class} ({styles.letterSpacing.value})
                  <br />
                  Font Weight: {styles.fontWeight.class} ({styles.fontWeight.value})
                  <br />
                  Paragraph Spacing: {styles.paragraphSpacing.class} ({styles.paragraphSpacing.value})
                </small>
              </div>
              <div className={className}>
                <div
                  style={{
                    fontSize: styles.fontSize.desktop,
                    lineHeight: styles.lineHeight.desktop,
                    letterSpacing: styles.letterSpacing.value,
                    fontWeight: styles.fontWeight.value,
                    marginBottom: styles.paragraphSpacing.value
                  }}
                >
                  {example}
                </div>
              </div>
            </div>
          ))}
        </div>
      </>
    );

  return (
    <div className="sb-typography-demo">
      <TokenSearchBar
        value={query}
        onChange={setQuery}
        placeholder="Search type classes (e.g. DisplayLarge, HeadingSmall, font-size-500)..."
        matches={matchCount}
        total={total}
      />

      {matchCount === 0 && (
        <div className="token-search-empty">
          No type classes match "<b>{query}</b>".
        </div>
      )}

      <h2>Type Scale Classes</h2>
      <div className="sb-type-scale-demo">
        {renderGroup('Display Types', display, 'Hello, Faclon! This is a display class example.')}
        {renderGroup('Heading Types', heading, 'Hello, Faclon! This is a heading classes example.')}
        {renderGroup('Body Types', body, 'Hello, Faclon! This is a body class example.')}
      </div>
    </div>
  );
};
