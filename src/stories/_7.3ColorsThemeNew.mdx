export const TokenSearchBar = ({ value, onChange, placeholder, matches, total }) => (
  <div className="token-search-wrap">
    <div className="token-search">
      <span className="token-search-icon" aria-hidden="true">{'\u{1F50D}'}</span>
      <input
        type="text"
        className="token-search-input"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        placeholder={placeholder || 'Search tokens...'}
      />
      {value && (
        <button
          type="button"
          className="token-search-clear"
          onClick={() => onChange('')}
          aria-label="Clear search"
        >
          {'✕'}
        </button>
      )}
    </div>
    {value && (
      <div className="token-search-meta">
        {matches} of {total} tokens match "<b>{value}</b>"
      </div>
    )}
  </div>
);

export const ThemeNewColorsDemo = () => {
  const [query, setQuery] = React.useState('');
  const q = query.trim().toLowerCase();

  const matches = (...fields) =>
    !q || fields.some((f) => f != null && String(f).toLowerCase().includes(q));

  const filterList = (list) =>
    list.filter((item) => matches(item.name, item.class));

  const filterGroups = (groupsObj) =>
    Object.entries(groupsObj)
      .map(([name, list]) => [name, filterList(list)])
      .filter(([, list]) => list.length > 0);

  const backgroundNewGroups = filterGroups(colors.backgroundNew);
  const textNewGroups = filterGroups(colors.textNew);
  const borderNewGroups = filterGroups(colors.borderNew);

  const countList = (groupsObj) =>
    Object.values(groupsObj).reduce((n, v) => n + v.length, 0);
  const countGroups = (groupEntries) =>
    groupEntries.reduce((n, [, list]) => n + list.length, 0);

  const total =
    countList(colors.backgroundNew) +
    countList(colors.textNew) +
    countList(colors.borderNew);

  const matchCount =
    countGroups(backgroundNewGroups) +
    countGroups(textNewGroups) +
    countGroups(borderNewGroups);

  const renderNewGrouped = (title, groupEntries) =>
    groupEntries.length > 0 && (
      <>
        <h2>{title}</h2>
        {groupEntries.map(([groupName, list]) => (
          <div key={groupName} className="color-section">
            <h3>{groupName.charAt(0).toUpperCase() + groupName.slice(1)}</h3>
            <div className="color-grid">
              {list.map(({ name, class: className }, index) => (
                <div key={index} className="color-item">
                  <div
                    className="color-preview"
                    style={{ backgroundColor: `var(${name})` }}
                  ></div>
                  <div className="color-info">
                    <div className="color-classes">
                      <h4>Class</h4>
                      <code>{className}</code>
                    </div>
                    <div className="color-variables">
                      <h4>Variable</h4>
                      <code>{name}</code>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          </div>
        ))}
      </>
    );

  return (
    <>
      <TokenSearchBar
        value={query}
        onChange={setQuery}
        placeholder="Search theme colors (e.g. background-brand-default, --text-error-default)..."
        matches={matchCount}
        total={total}
      />

      {matchCount === 0 && (
        <div className="token-search-empty">
          No theme color tokens match "<b>{query}</b>".
        </div>
      )}

      {(backgroundNewGroups.length > 0 ||
        textNewGroups.length > 0 ||
        borderNewGroups.length > 0) && (
        <div className="sb-section">
          <div className="sb-colors-demo">
            {renderNewGrouped('Background Classes (Newly added as on Dec 2025)', backgroundNewGroups)}
            {renderNewGrouped('Text Classes (Newly added as on Dec 2025)', textNewGroups)}
            {renderNewGrouped('Border Classes (Newly added as on Dec 2025)', borderNewGroups)}
          </div>
        </div>
      )}
    </>
  );
};
